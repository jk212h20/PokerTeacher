Four Color Cards

How to use in your project

/* 1. Copy cards.css and cards-ui.js into your project */

/* 2. Link them in your HTML */
<link rel="stylesheet" href="cards.css">
<script src="cards-ui.js"></script>

/* 3. Create cards in JS */
// From a shorthand string ("Ah", "Ts", "2d", "Kc")
const el = cardElementFromString("Ah");
container.appendChild(el);

// From a card object
const card = parseCard("Kh");
const el   = createCardElement(card);
container.appendChild(el);

// Render a whole hand at once
renderCards(document.getElementById("hand"), [
    parseCard("Ah"), parseCard("Kh"), parseCard("Qh"),
    parseCard("Jh"), parseCard("Th")
]);
